// // Navigation.swift // NeedU // // Created by Khaled El Abed on 12/10/2024. // import Foundation import SwiftUI import UIKit /// Modal / pushed navigation stack with centralized back-button and swipe-back interception. public final class NavigationController: UINavigationController, UINavigationControllerDelegate, UIGestureRecognizerDelegate { override public func viewDidLoad() { super.viewDidLoad() delegate = self interactivePopGestureRecognizer?.delegate = self customizeNavigationBar() } override public var preferredStatusBarStyle: UIStatusBarStyle { .default } // MARK: - Navigation bar private func customizeNavigationBar() { let color = UIColor.clear navigationBar.barStyle = .default navigationBar.isTranslucent = true navigationBar.setBackgroundImage(UIImage(), for: .default) navigationBar.shadowImage = UIImage() let navBarAppearance = UINavigationBarAppearance() navBarAppearance.configureWithOpaqueBackground() navBarAppearance.backgroundColor = color navBarAppearance.shadowColor = .clear navBarAppearance.titleTextAttributes = [.foregroundColor: Self.navigationTitleColor] navigationBar.standardAppearance = navBarAppearance navigationBar.scrollEdgeAppearance = navBarAppearance navigationBar.compactAppearance = navBarAppearance if #available(iOS 15.0, *) { navigationBar.compactScrollEdgeAppearance = navBarAppearance } } private static var navigationTitleColor: UIColor { if #available(iOS 17.0, *) { return UIColor(resource: .foregroundTitle) } return UIColor(named: "foregroundTitle", in: .main, compatibleWith: nil) ?? .label } // MARK: - Back interception @objc private func handleBackButtonTap() { performInterceptedPopIfNeeded() } private func performInterceptedPopIfNeeded() { guard viewControllers.count > 1 else { return } if let interceptor = topViewController as? NavigationBackIntercepting, !interceptor.onNavigationBack() { return } popViewController(animated: true) } private func configureNavigationBar(for viewController: UIViewController, animated: Bool) { let isRoot = viewController === viewControllers.first setNavigationBarHidden(false, animated: animated) if isRoot { // Keep any close button set by StandardRouter; only skip injecting "back". if viewController.navigationItem.leftBarButtonItem == nil { viewController.navigationItem.hidesBackButton = true } return } viewController.navigationItem.hidesBackButton = true if #available(iOS 14.0, *) { viewController.navigationItem.backButtonDisplayMode = .minimal } let backItem = UIBarButtonItem( image: UIImage(systemName: "chevron.left"), style: .plain, target: self, action: #selector(handleBackButtonTap) ) backItem.accessibilityLabel = NSLocalizedString("Back", comment: "Navigation back button") backItem.tintColor = Self.navigationTitleColor // `leftBarButtonItem` shows a tappable back control on *this* screen. // `backBarButtonItem` only customizes the back chevron on the *next* pushed screen. viewController.navigationItem.leftBarButtonItem = backItem if let popGesture = interactivePopGestureRecognizer { popGesture.isEnabled = true if popGesture.delegate !== self { popGesture.delegate = self } } } // MARK: - UINavigationControllerDelegate public func navigationController( _ navigationController: UINavigationController, willShow viewController: UIViewController, animated: Bool ) { configureNavigationBar(for: viewController, animated: animated) } public func navigationController( _ navigationController: UINavigationController, didShow viewController: UIViewController, animated: Bool ) { // UIHostingController / SwiftUI toolbar can reset bar items after `willShow`. configureNavigationBar(for: viewController, animated: false) } // MARK: - UIGestureRecognizerDelegate public func gestureRecognizerShouldBegin(_ gestureRecognizer: UIGestureRecognizer) -> Bool { guard gestureRecognizer === interactivePopGestureRecognizer else { return true } guard viewControllers.count > 1 else { return false } if let interceptor = topViewController as? NavigationBackIntercepting { return interceptor.onNavigationBack() } return true } }
